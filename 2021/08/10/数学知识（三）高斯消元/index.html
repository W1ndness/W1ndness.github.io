二、高斯消元

高斯消元的步骤：

  1. 对于增广矩阵 A~\tilde AA~​​ ，枚举每一列 ckc_kck​​​ ，初始 k=1k=1k=1
  2. 找到列 ckc_kck​ 未处理过的行（未处理过的方程）中绝对值最大的一元素，将该行换到最上面，即假设绝对值最大的一行为 rtr_trt​ ，则进行变换 rt↔rkr_t\leftrightarrow r_krt​↔rk​​​​
  3. 将该行第一个非零数（就是上面找到的绝对值最大的元素）变为 111​​​​​​​​​ ，相应的，其余各元素都乘以一个倍数。设当前第一个元素为 akja_{kj}akj​​​​​​​​ ，则对矩阵进行变换：rk×1akj\displaystyle r_k\times\frac 1 {a_{kj}}rk​×akj​1​​​​​​​​​​
  4. 用 akj=1a_{kj}=1akj​=1​​​​​ 将下面所有行的当前列消为 000​​​​​​​ ，即对矩阵进行变换 ∀i,−rk×aijakj+ri\forall i,\displaystyle-r_k\times\frac{a_{ij}}{a_{kj}}+r_i∀i,−rk​×akj​aij​​+ri​​​​​​
  5. k=k+1k=k+1k=k+1​ ，进行下一轮迭代，直至 k=nk=nk=n
/** Defination for Gauss-Jordan Elimination
* 	double a[N][N];		// store Equations
*	const double eps = 1e-6;	// error < eps see as 0
*	define func<gauss> return value:
*		return 0: has only one solution
* 		return 1: has infinite group solutions
*		return 2: has not solution
*/

int gauss() {
    int c, r;
    for (c = 0, r = 0; c < n; c++) {
        int t = r;
        for (int i = r; i < n; i++) {		// get max value from column<c>
            if (fabs(a[i][c]) > fabs(a[t][c])) t = i;
        }
        if (fabs(a[t][c]) < eps) continue;	// judge if it is 0
        for (int i = c; i <= n; i++) swap(a[t][i], a[r][i]);
        for (int i = n; i >= c; i--) a[r][i] /= a[r][c];
        for (int i = r + 1; i < n; i++) {
            if (fabs(a[i][c]) > eps) {
                for (int j = n; j >= c; j--) {
                    a[i][j] -= a[r][j] * a[i][c];
                }
            }
        }
        r++;
    }
    
    if (r < n) {
        for (int i = r; i < n; i++) {
            if (fabs(a[i][n]) > eps) return 2;
        }
        return 1;
    }
    
    for (int i = n - 1; i >= 0; i--) {
        for (int j = i + 1; j < n; j++) {
            a[i][n] -= a[i][j] * a[j][n];
        }
    }
    return 0;
}

#include <iostream>
using namespace std;

const int N = 110;

int n;
int a[N][N];

int gauss() {
    int c, r;
    for (c = 0, r = 0; c < n; c++) {
        int t = r;
        for (int i = r; i < n; i++) {
            if (a[i][c] == 1) {
                t = i;
                break;
            }
        }
        if (!a[t][c]) continue;
        for (int i = c; i < n + 1; i++) swap(a[r][i], a[t][i]);
        for (int i = r + 1; i < n; i++) {
            if (a[i][c]) {
                for (int j = n; j >= c; j--) {
                    a[i][j] ^= a[r][j];
                }
            }
        }
        r++;
    }
    if (r < n) {
        for (int i = r; i < n; i++) {
            if (a[i][n]) return 2;
        }
        return 1;
    }
    for (int i = n - 1; i >= 0; i--) {
        for (int j = i + 1; j < n; j++) {
            a[i][n] ^= a[i][j] & a[j][n];
        }
    }
    return 0;
}
