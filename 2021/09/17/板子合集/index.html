void quick_sort(int q[], int l, int r) {
    if (l >= r) return;
    int x = q[l + r >> 1];
    int i = l - 1, j = r + 1;	
      while (i < j) {
      do i++; while (q[i] < x);
      do j--; while (q[j] > x);
      if (i < j) swap(q[i], q[j]);
    }
    quick_sort(q, l, j);
    quick_sort(q, j + 1, r);
}

void down(int u) {
    int t = u;
    if (u * 2 <= cnt && h[u * 2] < h[t]) t = u * 2;
    if (u * 2 + 1 <= cnt && h[u * 2 + 1] < h[t]) t = u * 2 + 1;
    if (u != t) {
        swap(h[u], h[t]);
        down(t);
    }
}

void up(int u) {
    while (u / 2 && h[u / 2] > h[u]) {
        swap(h[u / 2], h[u]);

int dijkstra() {
    memset(dist, 0x3f, sizeof(dist));
    dist[1] = 0;
    for (int i = 0; i < n; i++) {	// 遍历所有点
        int t = -1;
        for (int j = 1; j <= n; j++) {	// 循环所有的点找到未确定最短路的、最小dist值的点
            if (!st[j]) && (t == -1 || dist[t] > dist[j]) t = j;
        }
        st[t] = true;

    }

}

int dijkstra() { // heap-dijkstra
    memset(dist, 0x3f, sizeof dist);
    dist[1] = 0;
    priority_queue<PII, vector<PII>, greater<PII>> heap;	// 小根堆 
    heap.push({0, 1});
    while (heap.size()) {
        auto t = heap.top(); heap.pop();
        int ver = t.second, distance = t.first;
        if (st[ver]) continue;	// redundancy 冗余
        else st[ver] = true;
        for (int i = h[ver]; i != -1; i = ne[i]) {
            int to = e[i];
            if (dist[to] > distance + w[i]) {
                dist[to] = distance + w[i];
                heap.push({dist[to], to});
            }
        }
    }
    if (dist[n] == 0x3f3f3f3f) return -1;
    else return dist[n];
}

    /* read in */
    memset(h, -1, sizeof h);
    /* read in edge info */
    int res = dijkstra();

bool dfs(int u, int c) {
    color[u] = c;

        int j = e[i];

// get result
LL res = 1;
for (auto &prime: primes) {
    int p = prime.first, a = prime.second;
    LL t = 1;
    while (a--) t = (t * p + 1) % mod;		// 秦九韶算法，得到公式中的每一个因式
    res = res * t % mod;
}

int getEuler(int a) {
    int res = a;
    for (int i = 2; i <= a / i; i++) {
        if (a % i == 0) {
            res = res / i * (i - 1);
            while (a % i == 0) a /= i;
        }
    }
    if (a > 1) res = res / a * (a - 1);
    return res;
}
/** Defination for linear sieve to get Euler
*	int primes[N], cnt;
*	int phi[N];
*	bool st[N];
*/

void getEulers(int n) {
    phi[1] = 1;
    for (int i = 2; i <= n; i++) {
        if (!st[i]) {
            primes[cnt++] = i;
            phi[i] = i - 1;
        }
        for (int j = 0; primes[j] <= n / i; j++) {
            st[primes[j] * i] = true;
            if (i % primes[j] == 0) {
                phi[primes[j] * i] = primes[j] * phi[i];
                break;
            }
            else phi[primes[j] * i] = (primes[j] - 1) * phi[i];
        }
    }
}

int C(int a, int b) {
    int res = 1;
    for (int i = 1, j = a; i <= b; i++, j--) {
        res = (LL)res * j % p;
        res = (LL)res * quickPow(i, p - 2) % p;
    }
    return res;
}

int lucas(LL a, LL b) {
    if (a < p && b < p) return C(a, b);
    return (LL)C(a % p, b % p) * lucas(a / p, b / p) % p;
}

int main() {
    /* read in */
    int res = 0;
    for (int i = 1; i < 1 << m; i++) {
        int t = 1, cnt = 0;
        for (int j = 0; j < m; j++) {
            if (i >> j & 1) {		// this set is chosen
                if ((LL)t * p[j] > n) {
                    t = -1;
                    break;
                }
                t *= p[j];	
                cnt++;
            }
        }
        if (t != -1) {
            if (cnt % 2) res += n / t;
            else res -= n / t;
        }
    }
    /* output */
    return 0;
}
