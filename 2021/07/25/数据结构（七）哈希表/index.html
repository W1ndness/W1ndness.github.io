数据结构（七）哈希表

一、哈希表的作用

哈希表是将值域范围比较大的一组数据映射到 0∼N0\sim N0∼N 上

哈希函数：h(x)∈[0,N]h(x)\in[0,N]h(x)∈[0,N]​​ ，哈希函数一般取 x\mod N​， NNN 一般取成质数且贴近 2k2^k2​k​​​

所以一般要先找到距离所开数组最大范围的最近的一个质数，将 NNN 设置成那个数

然而哈希函数可能会产生冲突，解决散列冲突的方式就是以下两种

二、哈希表的存储结构

注：以下例子假定 N=1e5N=1e5N=1e5

1.开放寻址法

开放寻址法一般要开二到三倍的数据范围大小的数组，一般需要开 2N∼3N2N\sim3N2N∼3N​​ 大的数组，在如果出现散列冲突，就继续向下寻找，直至寻找到一个空位将元素放入，就叫开放寻址法。

const int N = 200003, null = 0x3f3f3f3f;
int h[N];
memset(h, 0x3f, sizeof(h));

线性探查法：若某元素存在，则返回该元素的位置，若不存在，则返回他应该存储的一个空位置。

int find(int x) {
    int k = (x % N + N) % N;
    while (h[k] != null && h[k] != x) {
        k++;
        if (k == N) k = 0;
    }
    return k;
}

插入

void insert(int x) {
	int k = find(x);
    h[k] = x;
}

查询

bool find(int x) {
    int k = find(x);
    if (h[k] != null) return true;
    else return false;
}

2.拉链法

当哈希函数 h(x)h(x)h(x) 的值相同时，假定对于每一个 hashhashhash 值 h(x)h(x)h(x)​ ，都有一组值 x1,x2,⋯,xmx_1,x_2,\cdots,x_mx​1​​,x​2​​,⋯,x​m​​ 的哈希值是 h(x)h(x)h(x) 。那么就对这一系列值形成一个单链表，就叫拉链法。

const int N = 100003;	// 距离 1e5 最近的质数是 100003
int h[N];	// 拉链表的槽
int e[N], ne[N], idx;	// 单链表
memset(h, -1, sizeof(h));

插入

void insert(int x) {
	int k = (x % N + N) % N;	// 为了确保余数为正数
    e[idx] = x;
    ne[idx] = h[k];
    h[k] = idx++;	// 头插法插入单链表
}

查询

bool find(int x) {
    int k = (x % N + N) % N;
    for (int i = h[k]; i != -1; i = ne[i]) {
        if (e[i] == x) return true;
    }
    return false;
}

字符串前缀哈希法：

定义 {hn}\{h_n\}{h​n​​} 为字符串 strstrstr 的前缀哈希值，其中 hih_ih​i​​ 为前 iii 位的哈希值。

那么该如何定义某一前缀字符串的哈希值？

​ 假定字符串为 s1s2⋯sns_1s_2\cdots s_ns​1​​s​2​​⋯s​n​​ ，那么字符串的 hih_ih​i​​ 对应的前缀为 s1s2⋯sis_1s_2\cdots s_is​1​​s​2​​⋯s​i​​ ，定义字符串的数值为 valvalval ，我们把字符串转换成一个 ppp 进制的数。那么，字符串 s1s2⋯sis_1s_2\cdots s_is​1​​s​2​​⋯s​i​​ 对应的值为 val=s1×pi−1+s2×pi−2+⋯+si×p0val=s_1\times p^{i-1}+s_2\times p^{i-2}+\cdots+s_i\times p^0val=s​1​​×p​i−1​​+s​2​​×p​i−2​​+⋯+s​i​​×p​0​​ ，定义哈希函数为 h(x)=x\mod Q ，则最后得到的 h(val)h(val)h(val) 就是 hih_ih​i​​ 的值。显然，h(x)∈(0,Q)h(x)\in (0,Q)h(x)∈(0,Q)​

这里需要注意两点。首先，对于一个字符我们不能将其换算为 000​​ 。假如将 AAA​​ 映射成 000​​ ，那么 AAAAAA​​ 也就是 000​​ ，将会无法区分，这里我们采用其 ASCII 码去进行换算；第二，这里我们假设我们的哈希方法是不存在冲突的。一般经验值为 p=131/13331p=131/13331p=131/13331​​ 、 Q=264Q=2^{64}Q=2​64​​​​ 。

以上就是定义的字符串哈希方式。我们通过这样的定义方式就可以得到所有子串的哈希值。假定给出区间 [l,r][l,r][l,r] ，那么根据前缀和的思想，我们需要的是 hr,hl−1h_r,h_{l-1}h​r​​,h​l−1​​ 这两个数。对于字符串 strstrstr 内部而言，由于我们定义了要将字符串转化为一个 kkk 进制的数，则从 s1s_1s​1​​ 到 sns_ns​n​​ 为字符串的高位到低位。那么，我们再分别单独看两个子串，这两个前缀子串当中，s1s_1s​1​​ 是高位，而 sl−1,srs_{l-1},s_rs​l−1​​,s​r​​ 是低位。但两个子串相对而言，虽然他们的高位相同，但是，其实在数值上，以 s1s_1s​1​​ 为例，这里我们用前缀字符串哈希值来代表某一个前缀子串，那么 hrh_rh​r​​ 的 s1s_1s​1​​ 对应的值应当是 s1×pr−1s_1\times p^{r-1}s​1​​×p​r−1​​ ，而相对应的 hl−1h_{l-1}h​l−1​​ 中是 s1×pl−2s_1\times p^{l-2}s​1​​×p​l−2​​ 。可以看到，其实两个字符串的哈希值差了 pr−1−(l−2)=pr−l+1p^{r-1-(l-2)}=p^{r-l+1}p​r−1−(l−2)​​=p​r−l+1​​ 倍的。那么，如果想要进行 ”同等“ 的运算，我们就应当将小的哈希值字符串按位扩大相差的倍数才能进行运算。所以，我们就得到了计算区间 [l,r][l,r][l,r] 的子串哈希值的公式为：hl,r=hr−hl×pr−l+1(l≤r)h_{l,r}=h_r-h_l\times p^{r-l+1}(l\leq r)h​l,r​​=h​r​​−h​l​​×p​r−l+1​​(l≤r) 。这里需要说明的一点是，我们定义的哈希函数是 h(x)=x\mod Q ，这里并不是不进行取模运算了，我们用 unsigned long long 存储所有的哈希值，unsigned long long 的阈值是 2642^{64}2​64​​ ，所以如果溢出了范围，也就相当于是对 2642^{64}2​64​​​ 进行取模运算了。

仿照我们上面扩大倍数的思想，我们得到哈希值的方式也很简单。相邻两位上相差的倍数就是 ppp ，那么就有如下公式成立：hi=hi−1×p+sih_i=h_{i-1}\times p+s_ih​i​​=h​i−1​​×p+s​i​​ 。

typedef unsigned long long ULL;
const int P = 131;
char str[N];
ULL h[N], p[N];		// h stores hash values, p stores the pow of P

预处理

p[0] = 1;
for (int i = 1; i <= n; i++) {
    p[i] = p[i - 1] * P;
    h[i] = h[i - 1] * P + str[i];
}

获取某一区间内子串的哈希值

ULL get(int l, int r) {
    return h[r] - h[l - 1] * p[r - l + 1];
}

判断两个子串是否完全相同：如果两个子串的哈希值完全相同，那么就判定两者完全相同。因为对于任意的每一个不同的字符串而言，其哈希值是唯一的。

bool judge(int l1, int r1, int l2, int r2) {
    if (get(l1, r1) == get(l2, r2)) return true;
    else return false;
}

#include <iostream>
#include <cstring>
using namespace std;

const int N = 100003;

int h[N];
int e[N], ne[N], idx;

void insert(int x) {
    int k = (x % N + N) % N;
    e[idx] = x;
    ne[idx] = h[k];
    h[k] = idx++;
}

bool find(int x) {
    int k = (x % N + N) % N;
    for (int i = h[k]; i != -1; i = ne[i]) {
        if (e[i] == x) return true;
    }
    return false;
}

int main() {
    int n;
    memset(h, -1, sizeof(h));
    scanf("%d", &n);
    while (n--) {
        char op[2];
        int x;
        scanf("%s%d", op, &x);
        if (op[0] == 'I') insert(x);
        else {
            if (find(x)) cout << "Yes" << endl;
            else cout << "No" << endl;
        }
    }
    return 0;
}

#include <iostream>
using namespace std;

typedef unsigned long long ULL;

const int N = 100010, P = 131;

ULL h[N], p[N];
char str[N];

inline ULL gethash(int l, int r) {
    return h[r] - h[l - 1] * p[r - l + 1];
}

int main() {
    int n, m;
    scanf("%d%d", &n, &m);
    scanf("%s", str + 1);
    p[0] = 1;
    for (int i = 1; i <= n; i++) {
        h[i] = h[i - 1] * P + str[i];
        p[i] = p[i - 1] * P;
    }
    while (m--) {
        int l1, r1, l2, r2;
        scanf("%d%d%d%d", &l1, &r1, &l2, &r2);
        if (gethash(l1, r1) == gethash(l2, r2)) cout << "Yes" << endl;
        else cout << "No" << endl;
    }
    return 0;
}
def gethash(l, r):
    return (h[r] - h[l - 1] * p[r - l + 1]) % (1 << 64)

if __name__ == '__main__':
    n, m = map(int, input().split())
    P = 131
    s = ' ' + input()
    h = [0] * len(s)
    p = [0] * len(s)
    p[0] = 1
    for i in range(1, n + 1):
        h[i] = (h[i - 1] * P + ord(s[i])) % (1 << 64)
        p[i] = (p[i - 1] * P) % (1 << 64)
    for _ in range(m):
        l1, r1, l2, r2 = map(int, input().split())
        if gethash(l1, r1) == gethash(l2, r2):
            print("Yes")
        else:
            print("No")
